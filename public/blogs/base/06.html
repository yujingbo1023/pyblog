    # 1.集合（set）

    集合是一个 无序 、可变、不允许数据重复的容器。

    # 1.1 定义

    v1 = { 11, 22, 33, "root" }
    1

    • 无序，无法通过索引取值。

    • 可变，可以添加和删除元素。

      v1 = {11,22,33,44}

        import random
    
    result = {}
    
    user_list = ["root", "admin", "wc"]
    
    # 1. 生成一副扑克牌
    total_poke_list = [ ("小王",14),("大王",15) ]
color_list = ["红桃", "黑桃", "方片", "梅花"]
    num_list = []
for num in range(1, 14):
        num_list.append(num)
    for color in color_list:
        for num in num_list:
            item = (color, num,)
            total_poke_list.append(item)
    
    # 2. 发牌 -> ["root", "admin", "wc"]
    for user in user_list:
        # 给用户发第一张牌
        score = 0
        index = random.randint(0, len(total_poke_list) - 1)
        poke = total_poke_list.pop(index) # ("花色",值)
        # JQK表示 0.5 点
        value = poke[1]
        if poke[1] > 10:
            value = 0.5 
        score += value
        print("给{}发的牌：{}{}，此刻所有牌面值总和:{}".format(user, poke[0], poke[1], score))
        
        # 用户选择是否继续要
        while True:
            choice = input("是否继续要牌(Y/N)？")
            choice = choice.upper()

            if choice not in {"Y", "N"}:

                value = 0.5

            # 大于11点，则用户爆了且分值变为0
            if score > 11:
                print("用户{}爆了".format(user))
                score = 0
                break
    
        result[user] = score
